겜브료 2.6용 기본 쉐이더.   
  
말 그대로 기본 골격만 갖춘 쉐이더지만   
  
새로운 쉐이더 작업을 할 때는 은근히 필요합니다.   
  
만날 만들어 놓고 작업하다가 저장해 보면, 기초 쉐이더 모습은 없어진지 오래 (...)   
  
그래서 새로 만든김에 백업해 놓습니다.   
  
사진은 내 책상.  
  
모니터가 하나 더 생겨서 쉐이더 짜기 편해유.   
  
![](/assets/images/posts/20090803_141953_c0055803_4a76736b65595.jpg)  

//----------------------------------------------  
// Transformation Matrices  
//----------------------------------------------

float4x4 matWorldViewProj : WORLDVIEWPROJ;

//----------------------------------------------  
// Shader body  
//----------------------------------------------

struct VS\_INPUT  
{  
 float4 Pos      : POSITION0;

};

struct VS\_OUTPUT  
{  
 float4 Pos   : POSITION;

};

struct PS\_INPUT  
{

};

VS\_OUTPUT vs\_std(VS\_INPUT In)  
{  
 VS\_OUTPUT Out;

 Out.Pos = mul(In.Pos, matWorldViewProj);

 return Out;  
}

float4 ps\_std() : COLOR0  
{  
   
 float4 Final = float4(.0f, 1.0f,1.0f,1.0f);  
 return Final;  
}

//----------------------------------------------  
// Techniques  
//---------------------------------------------------------------------------//

technique jp\_gamebryo\_Baseshader  
<  
   string Description = "겜브리오용 베이스쉐이더";  
   string NBTMethod = "ATI";  
   int NBTSourceUV = 0;  
   bool UsesNiRenderState = true;  
   bool UsesNiLightState = false;  
>  
{  
 pass p0  
 {  
  VertexShader = compile vs\_3\_0 vs\_std();  
  PixelShader = compile ps\_3\_0 ps\_std();

  AlphaBlendEnable = true;  
  AlphaTestEnable = false;  
  AlphaRef = 150;  
  AlphaFunc = GREATER;  
  SrcBlend = SrcAlpha;  
  DestBlend = InvSrcAlpha;  
  CullMode = CW;  
    
  ZWriteEnable = true;  
   }  
}
