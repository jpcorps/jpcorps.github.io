![](/images/c0055803_5101b20b385f2.png)using System;

//Test 10번  
namespace Test10  
{  
    class HelloWorld  
    {  
        static void Main()  
        {  
            char Alphabet = 'A';  
            int number = 1;  
              
            //앞에 2번 돌리기   
            for (int i = 0; i < 2; i++)  
            {  
                  
                for (int j = 0; j < i; j++)  
                {  
                    Console.Write("  ");  
                }

                Console.Write("{0} ", Alphabet);

                for (int k=3; k > (i\*2); k--)  
                {  
                    Console.Write("{0} ",number);  
                    number++;

                }  
                Console.Write("{0} ", Alphabet);  
                Alphabet++;

                Console.WriteLine();  
            }

            //C는 무식하게 쓰기

            Console.WriteLine ("    {0}",Alphabet);  
            Alphabet++;

            //뒤에 2번 돌리기   
            for (int i = 2; i > 0; i--)  
            {

                for (int j = 1; j < i; j++)  
                {  
                    Console.Write("  ");  
                }

                Console.Write("{0} ",Alphabet);

                for (int k = 4 ; k >= (i\* 2); k--)  
                {  
                    Console.Write("{0} ",number);  
                    number++;

                }  
                Console.Write("{0} ",Alphabet);  
                Alphabet++;

                Console.WriteLine();  
            }

        }  
    }  
}
