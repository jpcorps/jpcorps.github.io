//테스트 9번  
namespace Test9  
{  
    class HelloWorld  
    {  
        static void Main()  
        {  
            int n = 1;

            // 앞에 3줄   
            for (int i = 0; i < 3; i++)  
            {  
                for (int j = 2; j > i; j--)  
                {  
                    Console.Write("   ");                      
                }

                for (float k = 0; k <= i; k += 0.5f)  
                {  
                    if (n < 9)  
                    {  
                        Console.Write("{0}  ", n);  
                    }  
                    else  
                    {  
                        Console.Write("{0} ", n);  
                    }  
                    n += 2;  
                }  
                Console.WriteLine();  
            }

            // 뒤에 2줄

            for (int i = 0; i < 2; i++)  
            {  
                for (int j = 0; j <= i; j++)  
                {  
                    Console.Write("  ");  
                }

                for (float k = 1; k >= i; k -= 0.5f)  
                {  
                    Console.Write("{0} ", n);  
                    n += 2;  
                }

                Console.WriteLine();

            }  
        }  
    }  
}
