utility callback\_test " box rotate"  
(  
 slider left\_box "Left" range:[-90,90,0] orient:#vertical offset:[-0,0]  
 slider right\_box "Right" range:[90,-90,0] orient:#vertical offset:[0,-0]  
 button resetv "Reset" width:140  
   
 checkbox lockon "Lock" align:#center  
   
on resetv pressed do  
(  
 $Rightbox.rotation.x\_rotation = $Leftbox.rotation.x\_rotation = 0  
 left\_box.value = right\_box.value = 0  
)

on left\_box changed val do   
(  
 $Leftbox.rotation.x\_rotation = -val  
 if lockon.checked do    
 (  
  right\_box.value = val  
  $Rightbox.rotation.x\_rotation = -val  
 )   
)  
on Right\_box changed val do   
(  
 $Rightbox.rotation.x\_rotation = -val  
)

)
