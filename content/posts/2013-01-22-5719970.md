namespace ConsoleApplication1  
{  
    class Program  
    {  
        static void Main(string[] args)  
        {  
            int n = 2;  
            int Resultnumber = 1;

            for (int i = 0; i < 5; i++)  
            {  
                  
                for (int j = -3; Math.Abs(j) >= (Math.Abs(n)+1); j++)  
                {  
                    Console.Write(Resultnumber);  
                                         
                }

                n -= 1;//공백을 증가했다가 감소시킴

                Console.WriteLine();  
            }

        }  
    }  
}
